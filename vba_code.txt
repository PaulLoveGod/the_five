=== MODULE: Up_dateChart ===
Sub UpdateChart()

On Error Resume Next

Dim ws As Worksheet
Dim Cht As ChartObject
Dim i As Long: Dim cEnd As Long: Dim cVol As Long
Dim pMax As Double: Dim pMin As Double: Dim vMax As Double: pMin = 1000000000
Dim sEnd As String: sEnd = "F:F"
Dim sVolume As String: sVolume = "G:G"

Set ws = ActiveSheet
Set Cht = ws.ChartObjects(1)

ws.Calculate

cEnd = ws.Range(sEnd).Column
cVol = ws.Range(sVolume).Column
    
For i = 1 To ws.UsedRange.Row + ws.UsedRange.Rows.Count - 1
    If ws.Cells(i, cEnd) > pMax And IsNumeric(ws.Cells(i, cEnd)) And ws.Cells(i, cEnd) <> "" Then pMax = ws.Cells(i, cEnd)
    If ws.Cells(i, cEnd) < pMin And IsNumeric(ws.Cells(i, cEnd)) And ws.Cells(i, cEnd) <> "" Then pMin = ws.Cells(i, cEnd)
    If ws.Cells(i, cVol) > vMax And IsNumeric(ws.Cells(i, cVol)) And ws.Cells(i, cVol) <> "" Then vMax = ws.Cells(i, cVol)
Next

Cht.Chart.Axes(xlValue).MaximumScale = pMax * 1.1
Cht.Chart.Axes(xlValue).MinimumScale = pMin * 0.7
Cht.Chart.Axes(xlValue, xlSecondary).MinimumScale = 0
Cht.Chart.Axes(xlValue, xlSecondary).MaximumScale = vMax * 2.5

Application.Calculation = xlCalculationManual

End Sub


=== MODULE: z_modNaverFinanceHistory ===
Option Explicit

Dim vaResult_Down_NaverFinanceHistory As Variant
Dim vaResult_Right_NaverFinanceHistory As Variant

Function NaverFinanceHistory(StockCode As Variant, Optional StartDate As Variant, Optional EndDate As Variant, _
                    Optional ColumnOrder As String = "시가, 고가, 저가, 종가, 거래량, 전일비", Optional return_array As Boolean = False, _
                    Optional IncludeHeader As Boolean = True)

' 변수 설정
Dim objHTTP As MSXML2.ServerXMLHTTP60                     ' XML.HTTP
Dim strHTML As String                                                   ' HTML 문서 텍스트
Dim oHTMLDoc As IHTMLDocument                                ' oHTML Document -> XML로 받아올 때는 oHTML로 받아온 뒤 HTML로 변경
Dim HTMLDoc As HTMLDocument                                   ' HTML Document
Dim Items As IHTMLElementCollection                              ' HTML items 개체
Dim Table As HTMLTable: Dim TableRow As HTMLTableRow: Dim TableCell As HTMLTableCell    ' HTML Table 개체
Dim dictElement As Dictionary                               ' 각 항목 받아올 Dictionary
Dim vaTemp As Variant:   Dim vaItems As Variant             ' 임시 변수
Dim vaResult As Variant                                     ' 출력변수

Dim sURL As String: sURL = "https://finance.naver.com/item/sise_day.nhn?code="
Dim URL As String

Dim lngPage As Long: Dim lngStart As Long: Dim lngEnd As Long
Dim strLast As String: Dim lngLast As Long: lngLast = 0
Dim i1 As Long: Dim i2 As Long
Dim varCell As Variant
Dim blninit As Boolean: blninit = True

Dim meWS As Worksheet: Dim meRng As Range: Dim strRtn As String

Dim r As Long: Dim c As Long                                ' For 반복문 위한 변수들
Dim i As Long: Dim j As Long: Dim o As Long: j = 1: i = 1: o = 1


'<------------- v1.2.1 수정
'출력 순서별 열 번호 지정
Dim x1 As Long: Dim x2 As Long: Dim x3 As Long: Dim x4 As Long: Dim x5 As Long: Dim x6 As Long: Dim x As Long: x = 3
Dim vaOrders As Variant: Dim vaOrder As Variant

vaOrders = Split(ColumnOrder, ",")
For Each vaOrder In vaOrders
    Select Case Trim(vaOrder)
        Case "종가" 'x1
            x1 = x: x = x + 1
        Case "전일비" 'x2
            x2 = x: x = x + 1
        Case "시가" 'x3
            x3 = x: x = x + 1
        Case "고가" 'x4
            x4 = x: x = x + 1
        Case "저가" 'x5
            x5 = x: x = x + 1
        Case "거래량" 'x6
            x6 = x: x = x + 1
        Case Else: GoTo EH_Null
    End Select
Next

Dim indexNo As Long: indexNo = 1
Dim MaxCol As Long: MaxCol = x - 2

' 최적화
Optimize_On

' 변수 세팅
Set oHTMLDoc = New HTMLDocument
Set objHTTP = New MSXML2.ServerXMLHTTP60
Set dictElement = New Dictionary

Set meRng = Application.Caller
Set meWS = meRng.Parent

' 오류 발생시 EH 이동 -> TIEMOUT 오류
On Error GoTo EH:

' 입력한 주식 종목의 마지막 페이지 번호 확인
URL = sURL & StockCode & "&page=" & 1
objHTTP.setTimeouts 3000, 3000, 3000, 3000
objHTTP.Open "GET", URL, False
objHTTP.setRequestHeader "User-Agent", "Mozilla/4.0 (compatible; MSIE 6.0; Windows NT 5.0)"
objHTTP.sEnd
        
With oHTMLDoc
    .Open
    .write objHTTP.responseText
    .Close
End With

Set HTMLDoc = oHTMLDoc

'<---1.2.4 수정 pgPR 없을 시 (신규상장), lngLast 페이지 1로 반환 -->

On Error Resume Next
strLast = HTMLDoc.getElementsByClassName("pgRR").Item(, 1).innerHTML
i1 = InStr(1, strLast, "page=")
i2 = InStr(1, strLast, ">")
lngLast = CLng(Mid(strLast, i1 + 5, i2 - i1 - 6))

If lngLast = 0 Then lngLast = 1

On Error GoTo EH_Null:

' 출력할 배열의 머릿글 생성
' <---1.2.3 수정 머릿글 포함여부 -->
If IncludeHeader = True Then
ReDim vaTemp(1 To MaxCol + 1)
    vaTemp(1) = i
    vaTemp(2) = "날짜"
    If x1 > 0 Then vaTemp(x1) = "종가"
    If x2 > 0 Then vaTemp(x2) = "전일비"
    If x3 > 0 Then vaTemp(x3) = "시가"
    If x4 > 0 Then vaTemp(x4) = "고가"
    If x5 > 0 Then vaTemp(x5) = "저가"
    If x6 > 0 Then vaTemp(x6) = "거래량"
    dictElement.Add i, Array(vaTemp, 1)
    i = i + 1
End If

' 시작 페이지 및 종료 페이지 셋업
If IsMissing(StartDate) And IsMissing(EndDate) Then
    lngStart = 1: lngEnd = 2: StartDate = 1: EndDate = Now
ElseIf Not (IsNumeric(StartDate) Or IsDate(StartDate) Or IsMissing(StartDate)) Or Not (IsNumeric(EndDate) Or IsDate(EndDate) Or IsMissing(EndDate)) Then
        NaverFinanceHistory = CVErr(xlErrValue): Exit Function
Else
    If Not IsMissing(StartDate) Then
        StartDate = CDate(StartDate)
        i1 = DateDiff("d", StartDate, Now)
        lngEnd = Application.WorksheetFunction.RoundUp(i1 / 10, 0)
        If lngEnd > lngLast Then lngEnd = lngLast
        
    Else
        StartDate = 1
        lngEnd = lngLast
    End If
    
    If Not IsMissing(EndDate) Then
        EndDate = CDate(EndDate)
        i1 = DateDiff("d", EndDate, Now)
        lngStart = Application.WorksheetFunction.RoundUp(i1 / 15, 0)
    Else
        EndDate = Now
        lngStart = 1
    End If
    
End If

' <-- 수정 v1.1.1 , lngEnd 또는 lgnStart 0 일 경우 1로 변경 -->
If lngEnd = 0 Then lngEnd = 1
If lngStart = 0 Then lngStart = 1

' 시작페이지 부터 종료페이지 까지 크롤링
For lngPage = lngStart To lngEnd

        URL = sURL & StockCode & "&page=" & lngPage
        
        ' 작성한 URL로 HTTP 요청 전달
        objHTTP.setTimeouts 3000, 3000, 3000, 3000
        objHTTP.Open "GET", URL, False
        objHTTP.setRequestHeader "User-Agent", "Mozilla/4.0 (compatible; MSIE 6.0; Windows NT 5.0)"
        objHTTP.sEnd
        
        ' 받아온 HTTP 문을 oHTML 문서에 작성
        With oHTMLDoc
            .Open
            .write objHTTP.responseText
            .Close
        End With
        
        ' oHTML -> HTML 변환
        Set HTMLDoc = oHTMLDoc
    
        On Error GoTo EH_Null:
        
        ' 받아온 HTML 문서에서 Table 개체 받아옴
        Set Items = HTMLDoc.getElementsByTagName("Table")
            
        ' 테이블 하나씩 돌아가며 index 된 테이블 선택
        o = 1
        For Each Table In Items
            If o = indexNo Then
                ' 테이블 row 하나씩 돌아가며
                    For Each TableRow In Table.Rows
                    ' 해당 row의 열 정보 담은 배열 생성 {열개수, 값1, 값2, ... }
                    varCell = TableRow.Cells.Item(0).innerText
                    If varCell <> "" And varCell <> "날짜" And varCell <> " " Then
                        If CDate(Replace(varCell, ".", "/")) >= StartDate And CDate(Replace(varCell, ".", "/")) <= EndDate Then
                            ReDim vaTemp(1 To MaxCol + 1)
                            vaTemp(1) = 7
                            vaTemp(2) = Format(CDate(Replace(TableRow.Cells.Item(0).innerText, ".", "/")), "yyyy-mm-dd")
                            If x1 > 0 Then vaTemp(x1) = CDbl(Replace(Trim(TableRow.Cells.Item(1).innerText), ",", ""))
                            ' 전일비 상승/하락 확인
                            varCell = TableRow.Cells.Item(2).innerHTML
                            If InStr(1, varCell, "상승") > 0 Or InStr(1, varCell, "ico_up") > 0 Then i2 = 1 Else i2 = -1
                            If x2 > 0 Then vaTemp(x2) = CDbl(Replace(Trim(ExtractNumbers(TableRow.Cells.Item(2).innerText)), ",", "")) * i2
                            If x3 > 0 Then vaTemp(x3) = CDbl(Replace(Trim(TableRow.Cells.Item(3).innerText), ",", ""))
                            If x4 > 0 Then vaTemp(x4) = CDbl(Replace(Trim(TableRow.Cells.Item(4).innerText), ",", ""))
                            If x5 > 0 Then vaTemp(x5) = CDbl(Replace(Trim(TableRow.Cells.Item(5).innerText), ",", ""))
                            If x6 > 0 Then vaTemp(x6) = CDbl(Replace(Trim(TableRow.Cells.Item(6).innerText), ",", ""))
                            ' Dict 에 배열 추가 Value는 해당 행 번호
                            dictElement.Add i, Array(vaTemp, 1)
                            i = i + 1
                        End If
                    End If
                Next
                Exit For
            End If
            o = o + 1
        Next
Next

' 출력할 배열의 C, R값 반환
r = dictElement.Count
c = MaxCol

' R = 0 (출력값 없을 시) NULL 오류 반환 후 종료
If r = 0 Then GoTo EH_Null:
    
' Dict Items 배열로 반환
vaItems = dictElement.Items

' 반환할 Data 배열 초기화
ReDim vaResult(1 To r, 1 To c)

' Dict Items 돌아가며 Data에 추가
For i = 0 To r - 1
    c = MaxCol
    For j = 2 To c + 1
        vaResult(i + 1, j - 1) = vaItems(i)(0)(j)
    Next
Next

' 오류 방지위해 Column 값 다시 설정
c = MaxCol

' 결과값 출력
If return_array = True Then
    NaverFinanceHistory = vaResult
    Optimize_Off
    Exit Function
Else
    ' 배열반환여부 FALSE일 경우 주변 셀에 값 뿌려준 뒤 함수 종료
    ' 오른쪽으로 출력해야 할 값이 있을 경우 vaResult_Right 배열 생성
    If c > 1 Then
        ReDim vaResult_Right_NaverFinanceHistory(1 To 1, 1 To c - 1)
        For j = 2 To c
            vaResult_Right_NaverFinanceHistory(1, j - 1) = vaResult(1, j)
        Next
    End If
    
    ' 아래쪽으로 출력해야 할 값이 있을 경우 vaResult_Down배열 생성
    If r > 1 Then
        ReDim vaResult_Down_NaverFinanceHistory(1 To r - 1, 1 To c)
        For i = 2 To r
            For j = 1 To c
                vaResult_Down_NaverFinanceHistory(i - 1, j) = vaResult(i, j)
            Next
        Next
    End If
    
    ' 배열 값을 Caller 주변으로 뿌려준 뒤 함수 종료
    If c > 1 Then
        strRtn = "WriteCellValue_Right_NaverFinanceHistory('" & meWS.Name & "'!" & meRng.Offset(0, 1).Resize(1, c - 1).Address(False, False) & ")"
        Evaluate (strRtn)
    End If
    
    If r > 1 Then
        strRtn = "WriteCellValue_Down_NaverFinanceHistory('" & meWS.Name & "'!" & meRng.Offset(1, 0).Resize(r - 1, c).Address(False, False) & ")"
        Evaluate (strRtn)
    End If
    
    NaverFinanceHistory = vaResult(1, 1)
    Optimize_Off
End If

Exit Function

EH:
    NaverFinanceHistory = "#TIMEOUT!"
    Optimize_Off
Exit Function

EH_Null:
    NaverFinanceHistory = "#NULL"
    Optimize_Off
    
End Function

Private Function DeleteCellValue(Range As Range)
    Range.Value = ""

Function ExtractNumbers(s As String) As String
    Dim result As String
    Dim i As Integer
    
    result = ""
    
    For i = 1 To Len(s)
        If IsNumeric(Mid(s, i, 1)) Then
            result = result & Mid(s, i, 1)
        End If
    Next i
    
    ExtractNumbers = result
End Function

Dim sVal As String

vArrs = getElementsByTag(sHTML, "option")

                For j = 1 To c

                Next
            Next

        ' 배열 값을 Caller 주변으로 뿌려준 뒤 함수 종료

            strRtn = "WriteCellValue_Down_GETSTOCKNO('" & meWS.Name & "'!" & meRng.Offset(1, 0).Resize(r - 1, c).Address(False, False) & ")"
            Evaluate (strRtn)

        GetStockNo = Right(vaResult(1, 1), Len(vaResult(1, 1)) - 1)
        Optimize_Off
    End If
Else
    If return_array = True Then
        GetStockNo = vaResult(1, 1)
    Else
        GetStockNo = Right(vaResult(1, 1), Len(vaResult(1, 1)) - 1)
    End If
End If

Exit Function

'########################################################
' 오류처리
'########################################################

EH:
    ' 배열반환여부 FALSE일 경우 주변 셀에 값 뿌려준 뒤 함수 종료
    ' 기존 값 삭제
    If return_array = False And meRng.Offset(1, 0).Value <> "" Then

            Evaluate (strRtn)

            Evaluate (strRtn)
        End With
    End If
    
    GetStockNo = "#TIMEOUT!"
    Optimize_Off
Exit Function

EH_Null:
    ' 배열반환여부 FALSE일 경우 주변 셀에 값 뿌려준 뒤 함수 종료
    ' 기존 값 삭제
        If return_array = False And meRng.Offset(1, 0).Value <> "" Then
        i = meRng.End(xlDown).Row
        With meWS
            strRtn = "DeleteCellValue('" & meWS.Name & "'!" & meRng.Offset(0, 1).Address(False, False) & ")"
            Evaluate (strRtn)
            strRtn = "DeleteCellValue('" & meWS.Name & "'!" & meRng.Offset(1, 0).Resize(i - meRng.Row, 2).Address(False, False) & ")"

    End If

Private Function getElementsByTag(sHTML, Tag As String) As Variant

Dim vArr As Variant
Dim v As Variant
Dim i As Long

vArr = Split(sHTML, "<" & Tag)

For i = LBound(vArr) + 1 To UBound(vArr)
    vArr(i) = "<" & Tag & Split(vArr(i), "</" & Tag & ">")(0) & "</" & Tag & ">"
Next

getElementsByTag = vArr

End Function

Private Function getAttribute(sHTML, Atts As String) As Variant

Dim i As Long
Dim val As String
Dim e1 As Long: Dim e2 As Long: Dim e As Long
i = InStr(1, sHTML, Atts)
e1 = InStr(i, sHTML, " "): If e1 = 0 Then e1 = Len(sHTML)
e2 = InStr(i, sHTML, ">"): If e2 = 0 Then e2 = Len(sHTML)
If e1 > e2 Then e = e2 Else e = e1

On Error Resume Next
val = Mid(sHTML, i + Len(Atts) + 1, e - i - Len(Atts) - 1)
If Left(val, 1) = """" Then val = Right(val, Len(val) - 1)
If Right(val, 1) = """" Then val = Left(val, Len(val) - 1)

getAttribute = val

End Function

Private Function getInnerText(sHTML) As Variant

Dim val As String
Dim i As Long: Dim e As Long
i = InStr(1, sHTML, ">")
e = InStr(i, sHTML, "<")

val = Mid(sHTML, i + 1, e - i - 1)

getInnerText = val

End Function

Private Function getAttribute(sHTML, Atts As String) As Variant

Dim i As Long
Dim val As String
Dim e1 As Long: Dim e2 As Long: Dim e As Long
i = InStr(1, sHTML, Atts)
e1 = InStr(i, sHTML, " "): If e1 = 0 Then e1 = Len(sHTML)
e2 = InStr(i, sHTML, ">"): If e2 = 0 Then e2 = Len(sHTML)
If e1 > e2 Then e = e2 Else e = e1

On Error Resume Next
val = Mid(sHTML, i + Len(Atts) + 1, e - i - Len(Atts) - 1)
If Left(val, 1) = """" Then val = Right(val, Len(val) - 1)
If Right(val, 1) = """" Then val = Left(val, Len(val) - 1)

getAttribute = val

End Function

Private Function getInnerText(sHTML) As Variant

Dim i As Long: Dim e As Long
i = InStr(1, sHTML, ">")
e = InStr(i, sHTML, "<")

val = Mid(sHTML, i + 1, e - i - 1)

getInnerText = val

End Function

Private Function getElementsByTag(sHTML, Tag As String) As Variant

Dim v As Variant

vArr = Split(sHTML, "<" & Tag)

For i = LBound(vArr) + 1 To UBound(vArr)
    vArr(i) = "<" & Tag & Split(vArr(i), "</" & Tag & ">")(0) & "</" & Tag & ">"
Next

getElementsByTag = vArr

End Function

Private Function getAttribute(sHTML, Atts As String) As Variant

Dim i As Long
Dim val As String
Dim e1 As Long: Dim e2 As Long: Dim e As Long
i = InStr(1, sHTML, Atts)
e1 = InStr(i, sHTML, " "): If e1 = 0 Then e1 = Len(sHTML)
e2 = InStr(i, sHTML, ">"): If e2 = 0 Then e2 = Len(sHTML)
If e1 > e2 Then e = e2 Else e = e1

On Error Resume Next
val = Mid(sHTML, i + Len(Atts) + 1, e - i - Len(Atts) - 1)
If Left(val, 1) = """" Then val = Right(val, Len(val) - 1)
If Right(val, 1) = """" Then val = Left(val, Len(val) - 1)

getAttribute = val

End Function

Private Function getInnerText(sHTML) As Variant

Dim val As String
Dim i As Long: Dim e As Long
i = InStr(1, sHTML, ">")
e = InStr(i, sHTML, "<")

val = Mid(sHTML, i + 1, e - i - 1)

getInnerText = val

End Function

Function GetHttp(URL As String, Optional formText As String, Optional isWinHttp As Boolean = False, Optional HTMLOnly As Boolean = False, Optional RequestHeader As Variant) As Variant
 
'###############################################################
'오빠두엑셀 VBA 사용자지정함수 (https://www.oppadu.com)
'▶ GetHttp 함수
'▶ 웹에서 데이터를 받아옵니다.
'▶ 인수 설명
'_____________URL                      : 데이터를 스크랩할 웹 페이지 주소입니다.
'_____________formText              : Encoding 된 FormText 형식으로 보내야 할 경우, Send String에 쿼리문을 추가합니다.
'_____________isWinHttp             : WinHTTP 로 요청할지 여부입니다. Redirect가 필요할 경우 True로 입력하여 WinHttp 요청을 전송합니다.
'_____________RequestHeader     : RequestHeader를 배열로 입력합니다. 반드시 짝수(한 쌍씩 이루어진) 개수로 입력되어야 합니다.
'▶ 사용 예제
'Dim HtmlResult As Object
'Set htmlResult = GetHttp("https://www.naver.com")
'msgbox htmlResult.body.innerHTML
'###############################################################
 
Dim oHTMLDoc As Object: Dim objHTTP As Object
Dim i As Long: Dim blnAgent As Boolean: blnAgent = False
Dim sUserAgent As String: sUserAgent = "Mozilla/5.0 (Linux; Android 6.0; Nexus 5 Build/MRA58N) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/86.0.4240.183 Mobile Safari/537.36"
 
Application.DisplayAlerts = False
 
If Left(URL, 4) <> "http" Then URL = "http://" & URL
 
Set oHTMLDoc = CreateObject("HtmlFile")
 
If isWinHttp = False Then
    Set objHTTP = CreateObject("MSXML2.ServerXMLHTTP")
Else
    Set objHTTP = CreateObject("WinHttp.WinHttpRequest.5.1")
End If
 
objHTTP.setTimeouts 3000, 3000, 3000, 3000
objHTTP.Open "POST", URL, False
If Not IsMissing(RequestHeader) Then
    Dim vRequestHeader As Variant
    For Each vRequestHeader In RequestHeader
        Dim uHeader As Long: Dim Lheader As Long: Dim steps As Long
        uHeader = UBound(vRequestHeader): Lheader = LBound(vRequestHeader)
        If (uHeader - Lheader) Mod 2 = 0 Then GetHttp = CVErr(xlValue): Exit Function
        For i = Lheader To uHeader Step 2
            If vRequestHeader(i) = "User-Agent" Then blnAgent = True
            objHTTP.setRequestHeader vRequestHeader(i), vRequestHeader(i + 1)
        Next
    Next
End If
If blnAgent = False Then objHTTP.setRequestHeader "User-Agent", sUserAgent
 
objHTTP.sEnd formText
 
If HTMLOnly = False Then
    With oHTMLDoc
        .Open
        .write objHTTP.responseText
        .Close
    End With
 
    Set GetHttp = oHTMLDoc
    Set oHTMLDoc = Nothing
    Set objHTTP = Nothing

Else
    
    GetHttp = objHTTP.responseText
    Set objHTTP = Nothing
    
End If
 
Application.DisplayAlerts = True
 
End Function

Sub stock()
    ' 변수 선언
    Dim bot As New WebDriver
    Dim link As String
    Dim stockprice As String

    On Error GoTo ErrorHandler ' 오류 처리 시작

    ' WebDriver 시작 (Headless 모드)
    bot.AddArgument "--headless"
    bot.Start "chrome"

    ' URL 가져오기
    link = "https://finance.yahoo.com/quote/" & Range("W9") & "/key-statistics/"
    If link = "" Then
        MsgBox "URL이 비어 있습니다. 확인 후 다시 실행하세요.", vbExclamation
        Exit Sub
    End If

    ' 새 창 열기
    bot.ExecuteScript "window.open(arguments[0]);", Array(link)
    bot.SwitchToNextWindow

    ' 요소가 로드될 때까지 대기
    Application.Wait Now + TimeValue("0:00:05")

    ' 주가 정보 가져오기
    On Error Resume Next ' XPath 오류 무시
    stockprice = bot.FindElementByXPath("//*[@id='nimbus-app']/section/section/section/article/section[1]/div[2]/div[1]/section/div/section[1]/div[1]/fin-streamer[1]").Text
    If stockprice = "" Then
        MsgBox "주가 정보를 가져올 수 없습니다. XPath를 확인하세요.", vbCritical
        GoTo Cleanup
    End If
    On Error GoTo 0 ' 오류 무시 해제

    ' 값 시트에 기록
    ActiveSheet.Range("BA8").Value = stockprice

Cleanup:
    ' WebDriver 종료
    bot.Quit
    Exit Sub

ErrorHandler:
    MsgBox "오류가 발생했습니다: " & Err.Description, vbCritical
    On Error Resume Next
    bot.Quit
End Sub

Sub HeadlessSeleniumExample()
    Dim bot As New WebDriver
    Dim element As WebElement
    Dim URL As String
    
    ' 브라우저 설정
    bot.Start "chrome", ""
    bot.AddArgument "--headless" ' Headless 모드 활성화
    bot.AddArgument "--disable-gpu" ' GPU 비활성화 (Windows 필요)
    bot.AddArgument "--no-sandbox" ' 리눅스 환경에서 필요할 수 있음
    
    ' URL로 이동
    URL = "https://finance.yahoo.com/quote/AAPL"
    bot.Get URL
    
    ' 특정 요소를 찾기
    On Error Resume Next
    Set element = bot.FindElementByCss(".Trsdu(0.3s) .Fw(b)")
    
    If Not element Is Nothing Then
        MsgBox "Apple's current stock price: " & element.Text
    Else
        MsgBox "Failed to retrieve stock price."
    End If
    
    ' 브라우저 닫기
    bot.Quit
End Sub
